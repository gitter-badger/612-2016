29 AnalogInput::AnalogInput(uint32_t channel) {
30  std::stringstream buf;
31  buf << "Analog Input " << channel;
32  Resource::CreateResourceObject(inputs, kAnalogInputs);
33 
34  if (!checkAnalogInputChannel(channel)) {
35  wpi_setWPIErrorWithContext(ChannelIndexOutOfRange, buf.str());
36  return;
37  }
38 
39  if (inputs->Allocate(channel, buf.str()) ==
40  std::numeric_limits<uint32_t>::max()) {
41  CloneError(*inputs);
42  return;
43  }
44 
45  m_channel = channel;
46 
47  void* port = getPort(channel);
48  int32_t status = 0;
49  m_port = initializeAnalogInputPort(port, &status);
50  wpi_setErrorWithContext(status, getHALErrorMessage(status));
51  freePort(port);
52 
53  LiveWindow::GetInstance()->AddSensor("AnalogInput", channel, this);
54  HALReport(HALUsageReporting::kResourceType_AnalogChannel, channel);
55 }

278 void AnalogInput::ResetAccumulator() {
279  if (StatusIsFatal()) return;
280  int32_t status = 0;
281  resetAccumulator(m_port, &status);
282  wpi_setErrorWithContext(status, getHALErrorMessage(status));
283 
284  if (!StatusIsFatal()) {
285  // Wait until the next sample, so the next call to GetAccumulator*()
286  // won't have old values.
287  const float sampleTime = 1.0f / GetSampleRate();
288  const float overSamples = 1 << GetOversampleBits();
289  const float averageSamples = 1 << GetAverageBits();
290  Wait(sampleTime * overSamples * averageSamples);
291  }
292 }
