378 void AxisCamera::Capture() {
379  int consecutiveErrors = 0;
380 
381  // Loop on trying to setup the camera connection. This happens in a background
382  // thread so it shouldn't effect the operation of user programs.
383  while (!m_done) {
384  std::string requestString =
385  "GET /mjpg/video.mjpg HTTP/1.1\n"
386  "User-Agent: HTTPStreamClient\n"
387  "Connection: Keep-Alive\n"
388  "Cache-Control: no-cache\n"
389  "Authorization: Basic RlJDOkZSQw==\n\n";
390  m_captureMutex.lock();
391  m_cameraSocket = CreateCameraSocket(requestString, consecutiveErrors > 5);
392  if (m_cameraSocket != -1) {
393  ReadImagesFromCamera();
394  consecutiveErrors = 0;
395  } else {
396  consecutiveErrors++;
397  }
398  m_captureMutex.unlock();
399  Wait(0.5);
400  }

406 void AxisCamera::ReadImagesFromCamera() {
407  char *imgBuffer = nullptr;
408  int imgBufferLength = 0;
409 
410  // TODO: these recv calls must be non-blocking. Otherwise if the camera
411  // fails during a read, the code hangs and never retries when the camera comes
412  // back up.
413 
414  int counter = 2;
415  while (!m_done) {
416  char initialReadBuffer[kMaxPacketSize] = "";
417  char intermediateBuffer[1];
418  char *trailingPtr = initialReadBuffer;
419  int trailingCounter = 0;
420  while (counter) {
421  // TODO: fix me... this cannot be the most efficient way to approach this,
422  // reading one byte at a time.
423  if (recv(m_cameraSocket, intermediateBuffer, 1, 0) == -1) {
424  wpi_setErrnoErrorWithContext("Failed to read image header");
425  close(m_cameraSocket);
426  return;
427  }
428  strncat(initialReadBuffer, intermediateBuffer, 1);
429  // trailingCounter ensures that we start looking for the 4 byte string
430  // after
431  // there is at least 4 bytes total. Kind of obscure.
432  // look for 2 blank lines (\r\n)
433  if (nullptr != strstr(trailingPtr, "\r\n\r\n")) {
434  --counter;
435  }
436  if (++trailingCounter >= 4) {
437  trailingPtr++;
438  }
439  }
440  counter = 1;
441  char *contentLength = strstr(initialReadBuffer, "Content-Length: ");
442  if (contentLength == nullptr) {
443  wpi_setWPIErrorWithContext(IncompatibleMode,
444  "No content-length token found in packet");
445  close(m_cameraSocket);
446  if (imgBuffer) delete[] imgBuffer;
447  return;
448  }
449  contentLength = contentLength + 16; // skip past "content length"
450  int readLength = atol(contentLength); // get the image byte count
451 
452  // Make sure buffer is large enough
453  if (imgBufferLength < readLength) {
454  if (imgBuffer) delete[] imgBuffer;
455  imgBufferLength = readLength + kImageBufferAllocationIncrement;
456  imgBuffer = new char[imgBufferLength];
457  if (imgBuffer == nullptr) {
458  imgBufferLength = 0;
459  continue;
460  }
461  }
462 
463  // Read the image data for "Content-Length" bytes
464  int bytesRead = 0;
465  int remaining = readLength;
466  while (bytesRead < readLength) {
467  int bytesThisRecv =
468  recv(m_cameraSocket, &imgBuffer[bytesRead], remaining, 0);
469  bytesRead += bytesThisRecv;
470  remaining -= bytesThisRecv;
471  }
472 
473  // Update image
474  {
475  std::lock_guard<priority_mutex> lock(m_imageDataMutex);
476 
477  m_imageData.assign(imgBuffer, imgBuffer + imgBufferLength);
478  m_freshImage = true;
479  }
480 
481  if (WriteParameters()) {
482  break;
483  }
484  }
485 
486  close(m_cameraSocket);
487 }

558 int AxisCamera::CreateCameraSocket(std::string const &requestString,
559  bool setError) {
560  struct addrinfo *address = nullptr;
561  int camSocket;
562 
563  /* create socket */
564  if ((camSocket = socket(AF_INET, SOCK_STREAM, 0)) == -1) {
565  if (setError)
566  wpi_setErrnoErrorWithContext("Failed to create the camera socket");
567  return -1;
568  }
569 
570  if (getaddrinfo(m_cameraHost.c_str(), "80", nullptr, &address) == -1) {
571  if (setError) {
572  wpi_setErrnoErrorWithContext("Failed to create the camera socket");
573  close(camSocket);
574  }
575  return -1;
576  }
577 
578  /* connect to server */
579  if (connect(camSocket, address->ai_addr, address->ai_addrlen) == -1) {
580  if (setError)
581  wpi_setErrnoErrorWithContext("Failed to connect to the camera");
582  freeaddrinfo(address);
583  close(camSocket);
584  return -1;
585  }
586 
587  freeaddrinfo(address);
588 
589  int sent = send(camSocket, requestString.c_str(), requestString.size(), 0);
590  if (sent == -1) {
591  if (setError)
592  wpi_setErrnoErrorWithContext("Failed to send a request to the camera");
593  close(camSocket);
594  return -1;
595  }
596 
597  return camSocket;
