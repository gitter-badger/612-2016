36 static int32_t sendMessageHelper(uint32_t messageID, const uint8_t *data,
37  uint8_t dataSize, int32_t period) {
38  static const uint32_t kTrustedMessages[] = {
39  LM_API_VOLT_T_EN, LM_API_VOLT_T_SET, LM_API_SPD_T_EN, LM_API_SPD_T_SET,
40  LM_API_VCOMP_T_EN, LM_API_VCOMP_T_SET, LM_API_POS_T_EN, LM_API_POS_T_SET,
41  LM_API_ICTRL_T_EN, LM_API_ICTRL_T_SET};
42 
43  int32_t status = 0;
44 
45  for (auto& kTrustedMessage : kTrustedMessages) {
46  if ((kFullMessageIDMask & messageID) == kTrustedMessage) {
47  uint8_t dataBuffer[8];
48  dataBuffer[0] = 0;
49  dataBuffer[1] = 0;
50 
51  // Make sure the data will still fit after adjusting for the token.
52  assert(dataSize <= 6);
53 
54  for (uint8_t j = 0; j < dataSize; j++) {
55  dataBuffer[j + 2] = data[j];
56  }
57 
58  FRC_NetworkCommunication_CANSessionMux_sendMessage(
59  messageID, dataBuffer, dataSize + 2, period, &status);
60 
61  return status;
62  }
63  }
64 
65  FRC_NetworkCommunication_CANSessionMux_sendMessage(messageID, data, dataSize,
66  period, &status);
67 
68  return status;
69 }

74 void CANJaguar::InitCANJaguar() {
75  m_safetyHelper = std::make_unique<MotorSafetyHelper>(this);
76 
77  bool receivedFirmwareVersion = false;
78  uint8_t dataBuffer[8];
79  uint8_t dataSize;
80 
81  // Request firmware and hardware version only once
82  requestMessage(CAN_IS_FRAME_REMOTE | CAN_MSGID_API_FIRMVER);
83  requestMessage(LM_API_HWVER);
84 
85  // Wait until we've gotten all of the status data at least once.
86  for (int i = 0; i < kReceiveStatusAttempts; i++) {
87  Wait(0.001);
88 
89  setupPeriodicStatus();
90  updatePeriodicStatus();
91 
92  if (!receivedFirmwareVersion &&
93  getMessage(CAN_MSGID_API_FIRMVER, CAN_MSGID_FULL_M, dataBuffer,
94  &dataSize)) {
95  m_firmwareVersion = unpackint32_t(dataBuffer);
96  receivedFirmwareVersion = true;
97  }
98 
99  if (m_receivedStatusMessage0 && m_receivedStatusMessage1 &&
100  m_receivedStatusMessage2 && receivedFirmwareVersion) {
101  break;
102  }
103  }
104 
105  if (!m_receivedStatusMessage0 || !m_receivedStatusMessage1 ||
106  !m_receivedStatusMessage2 || !receivedFirmwareVersion) {
107  wpi_setWPIErrorWithContext(JaguarMessageNotFound, "Status data not found");
108  }
109 
110  if (getMessage(LM_API_HWVER, CAN_MSGID_FULL_M, dataBuffer, &dataSize))
111  m_hardwareVersion = dataBuffer[0];
112 
113  if (m_deviceNumber < 1 || m_deviceNumber > 63) {
114  std::stringstream buf;
115  buf << "device number \"" << m_deviceNumber
116  << "\" must be between 1 and 63";
117  wpi_setWPIErrorWithContext(ParameterOutOfRange, buf.str());
118  return;
119  }
120 
121  if (StatusIsFatal()) return;
122 
123  // 3330 was the first shipping RDK firmware version for the Jaguar
124  if (m_firmwareVersion >= 3330 || m_firmwareVersion < 108) {
125  std::stringstream buf;
126  if (m_firmwareVersion < 3330) {
127  buf << "Jag #" << m_deviceNumber << " firmware (" << m_firmwareVersion
128  << ") is too old (must be at least version 108 "
129  "of the FIRST approved firmware)";
130  } else {
131  buf << "Jag #" << m_deviceNumber << " firmware (" << m_firmwareVersion
132  << ") is not FIRST approved (must be at least "
133  "version 108 of the FIRST approved firmware)";
134  }
135  wpi_setWPIErrorWithContext(JaguarVersionError, buf.str());
136  return;
137  }
138 
139  switch (m_controlMode) {
140  case kPercentVbus:
141  case kVoltage:
142  // No additional configuration required... start enabled.
143  EnableControl();
144  break;
145  default:
146  break;
147  }
148  HALReport(HALUsageReporting::kResourceType_CANJaguar, m_deviceNumber,
149  m_controlMode);
150  LiveWindow::GetInstance()->AddActuator("CANJaguar", m_deviceNumber, this);
151 }

244 void CANJaguar::Set(float outputValue, uint8_t syncGroup) {
245  uint32_t messageID;
246  uint8_t dataBuffer[8];
247  uint8_t dataSize;
248 
249  if (m_safetyHelper && !m_safetyHelper->IsAlive() && m_controlEnabled) {
250  EnableControl();
251  }
252 
253  if (m_controlEnabled) {
254  switch (m_controlMode) {
255  case kPercentVbus: {
256  messageID = LM_API_VOLT_T_SET;
257  if (outputValue > 1.0) outputValue = 1.0;
258  if (outputValue < -1.0) outputValue = -1.0;
259  dataSize = packPercentage(dataBuffer,
260  (m_isInverted ? -outputValue : outputValue));
261  } break;
262  case kSpeed: {
263  messageID = LM_API_SPD_T_SET;
264  dataSize = packFXP16_16(dataBuffer,
265  (m_isInverted ? -outputValue : outputValue));
266  } break;
267  case kPosition: {
268  messageID = LM_API_POS_T_SET;
269  dataSize = packFXP16_16(dataBuffer, outputValue);
270  } break;
271  case kCurrent: {
272  messageID = LM_API_ICTRL_T_SET;
273  dataSize = packFXP8_8(dataBuffer, outputValue);
274  } break;
275  case kVoltage: {
276  messageID = LM_API_VCOMP_T_SET;
277  dataSize =
278  packFXP8_8(dataBuffer, (m_isInverted ? -outputValue : outputValue));
279  } break;
280  default:
281  wpi_setWPIErrorWithContext(IncompatibleMode,
282  "The Jaguar only supports Current, Voltage, "
283  "Position, Speed, and Percent (Throttle) "
284  "modes.");
285  return;
286  }
287  if (syncGroup != 0) {
288  dataBuffer[dataSize] = syncGroup;
289  dataSize++;
290  }
291 
292  sendMessage(messageID, dataBuffer, dataSize, kSendMessagePeriod);
293 
294  if (m_safetyHelper) m_safetyHelper->Feed();
295  }
296 
297  m_value = outputValue;
298 
299  verify();
300 }

441 bool CANJaguar::getMessage(uint32_t messageID, uint32_t messageMask,
442  uint8_t *data, uint8_t *dataSize) const {
443  uint32_t targetedMessageID = messageID | m_deviceNumber;
444  int32_t status = 0;
445  uint32_t timeStamp;
446 
447  // Caller may have set bit31 for remote frame transmission so clear invalid
448  // bits[31-29]
449  targetedMessageID &= CAN_MSGID_FULL_M;
450 
451  // Get the data.
452  FRC_NetworkCommunication_CANSessionMux_receiveMessage(
453  &targetedMessageID, messageMask, data, dataSize, &timeStamp, &status);
454 
455  // Do we already have the most recent value?
456  if (status == ERR_CANSessionMux_MessageNotFound)
457  return false;
458  else
459  wpi_setErrorWithContext(status, "receiveMessage");
460 
461  return true;
462 }

501 void CANJaguar::updatePeriodicStatus() const {
502  uint8_t data[8];
503  uint8_t dataSize;
504 
505  // Check if a new bus voltage/output voltage/current/temperature message
506  // has arrived and unpack the values into the cached member variables
507  if (getMessage(LM_API_PSTAT_DATA_S0, CAN_MSGID_FULL_M, data, &dataSize)) {
508  m_mutex.lock();
509  m_busVoltage = unpackFXP8_8(data);
510  m_outputVoltage = unpackPercentage(data + 2) * m_busVoltage;
511  m_outputCurrent = unpackFXP8_8(data + 4);
512  m_temperature = unpackFXP8_8(data + 6);
513  m_mutex.unlock();
514 
515  m_receivedStatusMessage0 = true;
516  }
517 
518  // Check if a new position/speed message has arrived and do the same
519  if (getMessage(LM_API_PSTAT_DATA_S1, CAN_MSGID_FULL_M, data, &dataSize)) {
520  m_mutex.lock();
521  m_position = unpackFXP16_16(data);
522  m_speed = unpackFXP16_16(data + 4);
523  m_mutex.unlock();
524 
525  m_receivedStatusMessage1 = true;
526  }
527 
528  // Check if a new limits/faults message has arrived and do the same
529  if (getMessage(LM_API_PSTAT_DATA_S2, CAN_MSGID_FULL_M, data, &dataSize)) {
530  m_mutex.lock();
531  m_limits = data[0];
532  m_faults = data[1];
533  m_mutex.unlock();
534 
535  m_receivedStatusMessage2 = true;
536  }
537 }

544 void CANJaguar::verify() {
545  uint8_t dataBuffer[8];
546  uint8_t dataSize;
547 
548  // If the Jaguar lost power, everything should be considered unverified.
549  if (getMessage(LM_API_STATUS_POWER, CAN_MSGID_FULL_M, dataBuffer,
550  &dataSize)) {
551  bool powerCycled = (bool)dataBuffer[0];
552 
553  if (powerCycled) {
554  // Clear the power cycled bit
555  dataBuffer[0] = 1;
556  sendMessage(LM_API_STATUS_POWER, dataBuffer, sizeof(uint8_t));
557 
558  // Mark everything as unverified
559  m_controlModeVerified = false;
560  m_speedRefVerified = false;
561  m_posRefVerified = false;
562  m_neutralModeVerified = false;
563  m_encoderCodesPerRevVerified = false;
564  m_potentiometerTurnsVerified = false;
565  m_forwardLimitVerified = false;
566  m_reverseLimitVerified = false;
567  m_limitModeVerified = false;
568  m_maxOutputVoltageVerified = false;
569  m_faultTimeVerified = false;
570 
571  if (m_controlMode == kPercentVbus || m_controlMode == kVoltage) {
572  m_voltageRampRateVerified = false;
573  } else {
574  m_pVerified = false;
575  m_iVerified = false;
576  m_dVerified = false;
577  }
578 
579  // Verify periodic status messages again
580  m_receivedStatusMessage0 = false;
581  m_receivedStatusMessage1 = false;
582  m_receivedStatusMessage2 = false;
583 
584  // Remove any old values from netcomms. Otherwise, parameters are
585  // incorrectly marked as verified based on stale messages.
586  getMessage(LM_API_SPD_REF, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
587  getMessage(LM_API_POS_REF, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
588  getMessage(LM_API_SPD_PC, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
589  getMessage(LM_API_POS_PC, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
590  getMessage(LM_API_ICTRL_PC, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
591  getMessage(LM_API_SPD_IC, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
592  getMessage(LM_API_POS_IC, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
593  getMessage(LM_API_ICTRL_IC, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
594  getMessage(LM_API_SPD_DC, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
595  getMessage(LM_API_POS_DC, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
596  getMessage(LM_API_ICTRL_DC, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
597  getMessage(LM_API_CFG_BRAKE_COAST, CAN_MSGID_FULL_M, dataBuffer,
598  &dataSize);
599  getMessage(LM_API_CFG_ENC_LINES, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
600  getMessage(LM_API_CFG_POT_TURNS, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
601  getMessage(LM_API_CFG_LIMIT_MODE, CAN_MSGID_FULL_M, dataBuffer,
602  &dataSize);
603  getMessage(LM_API_CFG_LIMIT_FWD, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
604  getMessage(LM_API_CFG_LIMIT_REV, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
605  getMessage(LM_API_CFG_MAX_VOUT, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
606  getMessage(LM_API_VOLT_SET_RAMP, CAN_MSGID_FULL_M, dataBuffer, &dataSize);
607  getMessage(LM_API_VCOMP_COMP_RAMP, CAN_MSGID_FULL_M, dataBuffer,
608  &dataSize);
609  getMessage(LM_API_CFG_FAULT_TIME, CAN_MSGID_FULL_M, dataBuffer,
610  &dataSize);
611  }
612  } else {
613  requestMessage(LM_API_STATUS_POWER);
614  }
615 
616  // Verify that any recently set parameters are correct
617  if (!m_controlModeVerified && m_controlEnabled) {
618  if (getMessage(LM_API_STATUS_CMODE, CAN_MSGID_FULL_M, dataBuffer,
619  &dataSize)) {
620  ControlMode mode = (ControlMode)dataBuffer[0];
621 
622  if (m_controlMode == mode)
623  m_controlModeVerified = true;
624  else
625  // Enable control again to resend the control mode
626  EnableControl();
627  } else {
628  // Verification is needed but not available - request it again.
629  requestMessage(LM_API_STATUS_CMODE);
630  }
631  }
632 
633  if (!m_speedRefVerified) {
634  if (getMessage(LM_API_SPD_REF, CAN_MSGID_FULL_M, dataBuffer, &dataSize)) {
635  uint8_t speedRef = dataBuffer[0];
636 
637  if (m_speedReference == speedRef)
638  m_speedRefVerified = true;
639  else
640  // It's wrong - set it again
641  SetSpeedReference(m_speedReference);
642  } else {
643  // Verification is needed but not available - request it again.
644  requestMessage(LM_API_SPD_REF);
645  }
646  }
647 
648  if (!m_posRefVerified) {
649  if (getMessage(LM_API_POS_REF, CAN_MSGID_FULL_M, dataBuffer, &dataSize)) {
650  uint8_t posRef = dataBuffer[0];
651 
652  if (m_positionReference == posRef)
653  m_posRefVerified = true;
654  else
655  // It's wrong - set it again
656  SetPositionReference(m_positionReference);
657  } else {
658  // Verification is needed but not available - request it again.
659  requestMessage(LM_API_POS_REF);
660  }
661  }
662 
663  if (!m_pVerified) {
664  uint32_t message = 0;
665 
666  if (m_controlMode == kSpeed)
667  message = LM_API_SPD_PC;
668  else if (m_controlMode == kPosition)
669  message = LM_API_POS_PC;
670  else if (m_controlMode == kCurrent)
671  message = LM_API_ICTRL_PC;
672  else {
673  wpi_setWPIErrorWithContext(
674  IncompatibleMode,
675  "PID constants only apply in Speed, Position, and Current mode");
676  return;
677  }
678 
679  if (getMessage(message, CAN_MSGID_FULL_M, dataBuffer, &dataSize)) {
680  double p = unpackFXP16_16(dataBuffer);
681 
682  if (FXP16_EQ(m_p, p))
683  m_pVerified = true;
684  else
685  // It's wrong - set it again
686  SetP(m_p);
687  } else {
688  // Verification is needed but not available - request it again.
689  requestMessage(message);
690  }
691  }
692 
693  if (!m_iVerified) {
694  uint32_t message = 0;
695 
696  if (m_controlMode == kSpeed)
697  message = LM_API_SPD_IC;
698  else if (m_controlMode == kPosition)
699  message = LM_API_POS_IC;
700  else if (m_controlMode == kCurrent)
701  message = LM_API_ICTRL_IC;
