134 void CANTalon::Set(float value, uint8_t syncGroup) {
135  /* feed safety helper since caller just updated our output */
136  m_safetyHelper->Feed();
137  if (m_controlEnabled) {
138  m_setPoint = value; /* cache set point for GetSetpoint() */
139  CTR_Code status = CTR_OKAY;
140  switch (m_controlMode) {
141  case CANSpeedController::kPercentVbus: {
142  m_impl->Set(m_isInverted ? -value : value);
143  status = CTR_OKAY;
144  } break;
145  case CANSpeedController::kFollower: {
146  status = m_impl->SetDemand((int)value);
147  } break;
148  case CANSpeedController::kVoltage: {
149  // Voltage is an 8.8 fixed point number.
150  int volts = int((m_isInverted ? -value : value) * 256);
151  status = m_impl->SetDemand(volts);
152  } break;
153  case CANSpeedController::kSpeed:
154  /* if the caller has provided scaling info, apply it */
155  status = m_impl->SetDemand(ScaleVelocityToNativeUnits(m_feedbackDevice, m_isInverted ? -value : value));
156  break;
157  case CANSpeedController::kPosition:
158  status = m_impl->SetDemand(ScaleRotationsToNativeUnits(m_feedbackDevice, value));
159  break;
160  case CANSpeedController::kCurrent: {
161  double milliamperes = (m_isInverted ? -value : value) * 1000.0; /* mA*/
162  status = m_impl->SetDemand(milliamperes);
163  } break;
164  case CANSpeedController::kMotionProfile: {
165  status = m_impl->SetDemand((int)value);
166  } break;
167  default:
168  wpi_setWPIErrorWithContext(
169  IncompatibleMode,
170  "The CAN Talon does not support this control mode.");
171  break;
172  }
173  if (status != CTR_OKAY) {
174  wpi_setErrorWithContext(status, getHALErrorMessage(status));
175  }
176 
177  status = m_impl->SetModeSelect(m_sendMode);
178 
179  if (status != CTR_OKAY) {
180  wpi_setErrorWithContext(status, getHALErrorMessage(status));
181  }
182  }
183 }

764 CANTalon::FeedbackDeviceStatus CANTalon::IsSensorPresent(FeedbackDevice feedbackDevice)const
765 {
766  FeedbackDeviceStatus retval = FeedbackStatusUnknown;
767  int param;
768  /* detecting sensor health depends on which sensor caller cares about */
769  switch (feedbackDevice) {
770  case QuadEncoder:
771  case AnalogPot:
772  case AnalogEncoder:
773  case EncRising:
774  case EncFalling:
775  /* no real good way to tell if these sensor
776  are actually present so return status unknown. */
777  break;
778  case PulseWidth:
779  case CtreMagEncoder_Relative:
780  case CtreMagEncoder_Absolute:
781  /* all of these require pulse width signal to be present. */
782  CTR_Code status = m_impl->IsPulseWidthSensorPresent(param);
783  if (status != CTR_OKAY) {
784  /* we're not getting status info, signal unknown status */
785  } else {
786  /* param is updated */
787  if (param) {
788  /* pulse signal is present, sensor must be working since it always
789  generates a pulse waveform.*/
790  retval = FeedbackStatusPresent;
791  } else {
792  /* no pulse present, sensor disconnected */
793  retval = FeedbackStatusNotPresent;
794  }
795  }
796  break;
797  }
798  return retval;
799 }

1499 double CANTalon::GetNativeUnitsPerRotationScalar(FeedbackDevice devToLookup)const
1500 {
1501  bool scalingAvail = false;
1502  CTR_Code status = CTR_OKAY;
1503  double retval = 0;
1504  switch (devToLookup) {
1505  case QuadEncoder:
1506  { /* When caller wants to lookup Quadrature, the QEI may be in 1x if the selected feedback is edge counter.
1507  * Additionally if the quadrature source is the CTRE Mag encoder, then the CPR is known.
1508  * This is nice in that the calling app does not require knowing the CPR at all.
1509  * So do both checks here.
1510  */
1511  int32_t qeiPulsePerCount = 4; /* default to 4x */
1512  switch (m_feedbackDevice) {
1513  case CtreMagEncoder_Relative:
1514  case CtreMagEncoder_Absolute:
1515  /* we assume the quadrature signal comes from the MagEnc,
1516  of which we know the CPR already */
1517  retval = kNativePwdUnitsPerRotation;
1518  scalingAvail = true;
1519  break;
1520  case EncRising: /* Talon's QEI is setup for 1x, so perform 1x math */
1521  case EncFalling:
1522  qeiPulsePerCount = 1;
1523  break;
1524  case QuadEncoder: /* Talon's QEI is 4x */
1525  default: /* pulse width and everything else, assume its regular quad use. */
1526  break;
1527  }
1528  if (scalingAvail) {
1529  /* already deduced the scalar above, we're done. */
1530  } else {
1531  /* we couldn't deduce the scalar just based on the selection */
1532  if (0 == m_codesPerRev) {
1533  /* caller has never set the CPR. Most likely caller
1534  is just using engineering units so fall to the
1535  bottom of this func.*/
1536  } else {
1537  /* Talon expects PPR units */
1538  retval = qeiPulsePerCount * m_codesPerRev;
1539  scalingAvail = true;
1540  }
1541  }
1542  } break;
1543  case EncRising:
1544  case EncFalling:
1545  if (0 == m_codesPerRev) {
1546  /* caller has never set the CPR. Most likely caller
1547  is just using engineering units so fall to the
1548  bottom of this func.*/
1549  } else {
1550  /* Talon expects PPR units */
1551  retval = 1 * m_codesPerRev;
1552  scalingAvail = true;
1553  }
1554  break;
1555  case AnalogPot:
1556  case AnalogEncoder:
1557  if (0 == m_numPotTurns) {
1558  /* caller has never set the CPR. Most likely caller
1559  is just using engineering units so fall to the
1560  bottom of this func.*/
1561  } else {
1562  retval = (double)kNativeAdcUnitsPerRotation / m_numPotTurns;
1563  scalingAvail = true;
1564  }
1565  break;
1566  case CtreMagEncoder_Relative:
1567  case CtreMagEncoder_Absolute:
1568  case PulseWidth:
1569  retval = kNativePwdUnitsPerRotation;
1570  scalingAvail = true;
1571  break;
1572  }
1573  /* handle any detected errors */
1574  if (status != CTR_OKAY) {
1575  wpi_setErrorWithContext(status, getHALErrorMessage(status));
1576  }
1577  /* if scaling information is not possible, signal caller
1578  by returning zero */
1579  if (false == scalingAvail)
1580  retval = 0;
1581  return retval;
1582 }
