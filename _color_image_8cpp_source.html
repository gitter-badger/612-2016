155 MonoImage *ColorImage::ExtractColorPlane(ColorMode mode, int planeNumber) {
156  auto result = new MonoImage();
157  if (m_imaqImage == nullptr) wpi_setWPIError(NullParameter);
158  int success = imaqExtractColorPlanes(
159  m_imaqImage, mode, (planeNumber == 1) ? result->GetImaqImage() : nullptr,
160  (planeNumber == 2) ? result->GetImaqImage() : nullptr,
161  (planeNumber == 3) ? result->GetImaqImage() : nullptr);
162  wpi_setImaqErrorWithContext(success, "Imaq ExtractColorPlanes failed");
163  return result;
164 }
