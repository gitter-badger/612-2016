45  uint32_t lineNumber) const {
46  std::string err;
47  int errNo = errno;
48  if (errNo == 0) {
49  err = "OK: ";
50  err += contextMessage;
51  } else {
52  char buf[256];
53  snprintf(buf, 256, "%s (0x%08X): %.*s", strerror(errNo), errNo,
54  contextMessage.size(), contextMessage.data());
55  err = buf;
56  }
