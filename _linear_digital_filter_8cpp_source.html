121 double LinearDigitalFilter::Get() const {
122  double retVal = 0.0;
123 
124  // Calculate the new value
125  for (unsigned int i = 0; i < m_inputGains.size(); i++) {
126  retVal += m_inputs[i] * m_inputGains[i];
127  }
128  for (unsigned int i = 0; i < m_outputGains.size(); i++) {
129  retVal -= m_outputs[i] * m_outputGains[i];
130  }
131 
132  return retVal;
133 }

148 double LinearDigitalFilter::PIDGet() {

151  // Rotate the inputs
152  m_inputs.PushFront(PIDGetSource());
153 
154  // Calculate the new value
155  for (unsigned int i = 0; i < m_inputGains.size(); i++) {
156  retVal += m_inputs[i] * m_inputGains[i];
157  }
158  for (unsigned int i = 0; i < m_outputGains.size(); i++) {
159  retVal -= m_outputs[i] * m_outputGains[i];
160  }
161 
162  // Rotate the outputs
163  m_outputs.PushFront(retVal);
164 
165  return retVal;
