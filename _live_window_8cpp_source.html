37 void LiveWindow::SetEnabled(bool enabled) {
38  if (m_enabled == enabled) return;
39  if (enabled) {
40  if (m_firstTime) {
41  InitializeLiveWindowComponents();
42  m_firstTime = false;
43  }
44  m_scheduler->SetEnabled(false);
45  m_scheduler->RemoveAll();
46  for (auto& elem : m_components) {
47  elem.first->StartLiveWindowMode();
48  }
49  } else {
50  for (auto& elem : m_components) {
51  elem.first->StopLiveWindowMode();
52  }
53  m_scheduler->SetEnabled(true);
54  }
55  m_enabled = enabled;
56  m_statusTable->PutBoolean("LW Enabled", m_enabled);
57 }
