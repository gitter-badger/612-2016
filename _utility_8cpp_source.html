182 static std::string demangle(char const *mangledSymbol) {
183  char buffer[256];
184  size_t length;
185  int status;
186 
187  if (sscanf(mangledSymbol, "%*[^(]%*[(]%255[^)+]", buffer)) {
188  char *symbol = abi::__cxa_demangle(buffer, nullptr, &length, &status);
189  if (status == 0) {
190  return symbol;
191  } else {
192  // If the symbol couldn't be demangled, it's probably a C function,
193  // so just return it as-is.
194  return buffer;
195  }
196  }
197 
198  // If everything else failed, just return the mangled symbol
199  return mangledSymbol;
200 }

206 std::string GetStackTrace(uint32_t offset) {
207  void *stackTrace[128];
208  int stackSize = backtrace(stackTrace, 128);
209  char **mangledSymbols = backtrace_symbols(stackTrace, stackSize);
210  std::stringstream trace;
211 
212  for (int i = offset; i < stackSize; i++) {
213  // Only print recursive functions once in a row.
214  if (i == 0 || stackTrace[i] != stackTrace[i - 1]) {
215  trace << "\tat " << demangle(mangledSymbols[i]) << std::endl;
216  }
217  }
218 
219  free(mangledSymbols);
220 
221  return trace.str();
222 }
